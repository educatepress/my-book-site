// ── Types ──
interface BankEntry {
  id: string;
  status: 'pending' | 'scripted' | 'rendered';
  title: string;
  hook: string;
  text: string;
  videoKeyword: string;
  themeArea: string;
  type: string;
  evidence?: { summary: string; paperTitle: string; pmidOrDoi: string; source: string };
  trend?: { platform: string; description: string; evidenceBacked: boolean; reference: string };
  [key: string]: unknown;
}

interface ScriptOutput {
  hook: string;
  act2_openLoop: string;
  act3_coreInsight: string;
  act4_midrollReHook: string;
  act5_payoffCta: string;
  videoKeyword: string;
  emphasisWords: string[];
  keyTakeaway: string;
  thumbnailText: string;
}

// ── System Prompt (5-Act Retention Architecture + Medical Authority) ──
const SYSTEM_PROMPT = `You are a world-class short-form video scriptwriter specializing in fertility medicine (TTC / infertility / reproductive health).

You write scripts for a fertility doctor who creates Instagram Reels and TikTok videos targeted at English-speaking women aged 20-35 who are trying to conceive.

## 🎯 PRIMARY GOAL: Maximize Retention Rate
Every decision you make should optimize for:
1. **1-3 second retention** (the "scroll-stop" moment)
2. **Watch-through rate** (keep viewers until the end)
3. **Replay value** (make them want to watch again or share)

## 📐 Script Structure: 5-Act Micro-Architecture (20 seconds)

### Act 1: PATTERN INTERRUPT — THE SCROLL-STOPPER (0-2 sec, ~8-12 words)
The single most important line. It decides if the viewer stays or leaves.

MANDATORY RULES:
- Must make the viewer think: "Wait—that's about ME." within 0.5 seconds
- Must include WHO this is for (age, situation, identity) — be SPECIFIC
- Must create emotional charge (surprise, curiosity, or slight provocation)
- NEVER use generic openers (see BLACKLIST below)

HOOK FORMULA TEMPLATES (pick the best one for the topic):
1. IDENTITY + PROVOCATIVE CLAIM: "Trying to conceive after 35? Stop doing this."
2. MYTH-BUST OPENER: "Everything you've been told about AMH is wrong."
3. DIRECT CHALLENGE: "Your partner's phone might be ruining your chances."
4. DOCTOR'S SECRET: "Your fertility doctor won't tell you this."
5. EMOTIONAL MIRROR: "If your AMH result made you cry, watch this."
6. COUNTER-INTUITIVE: "The supplement everyone takes? It might not be enough."

BLACKLIST (NEVER USE these weak patterns):
❌ "Did you know..."
❌ "In this video..."
❌ "Here's a fact about..."
❌ Single noun questions like "Sperm health?" or "AMH levels?"
❌ Generic phrases without a target audience ("You might be surprised")
❌ Starting with the answer/solution name (e.g., "CoQ10 is...")

### Act 2: OPEN LOOP — THE CURIOSITY TRAP (2-5 sec, ~10-15 words)
The viewer stayed for the hook. Now LOCK THEM IN by creating unbearable curiosity.

MANDATORY RULES:
- NEVER reveal the main keyword, solution, or answer here
- Must create a "knowledge gap" — the viewer NEEDS to keep watching to close it
- Use specificity to tease: numbers, categories, or conditions — but NOT the answer

OPEN LOOP FORMULA TEMPLATES:
1. MYSTERY INGREDIENT: "And there's ONE specific compound that changes everything."
2. HIDDEN CONSEQUENCE: "This everyday habit could be costing you months of trying."
3. DOCTOR'S OMISSION: "Here's what most clinics don't explain during your consult."
4. NUMBERED TEASE: "There are three things you can do today. The last one is critical."
5. STAKES ESCALATION: "And if you don't know about this, you might be wasting precious time."

BLACKLIST (NEVER USE):
❌ "And you might be surprised by the findings." (too generic)
❌ "Let me tell you..." (too passive)
❌ "Here's something interesting..." (no tension)
❌ Starting with the answer (e.g., "CoQ10 can help...")

### Act 3: CORE INSIGHT (5-13 sec, ~25-30 words)
The main educational payload. NOW reveal the answer.
- Use a simple analogy to explain complex concepts (e.g., "basically the battery pack")
- Include specific data (numbers create credibility): "28,000 women"
- Keep sentences SHORT (8 words max per sentence)

### Act 4: MIDROLL RE-HOOK (13-16 sec, ~10-15 words)
This is where viewers typically drop off. Re-engage with a TWIST:
- Start with "But here's what most people miss..."
- Or: "However, there's a catch..."
- Introduce a caveat, timing tip, or counter-intuitive point
- This must feel like NEW information, not a repeat

### Act 5: PAYOFF + FUNNEL CTA (16-20 sec, ~10-15 words)
Drive BOTH algorithmic engagement AND audience funnel. MUST include:
- An **algorithmic action**: "Save this" or "Share with someone trying to conceive" (boosts saves/shares)
- A **funnel trigger**: "Comment GUIDE for my free fertility checklist"
- Example: "Save this — and comment GUIDE for the full fertility checklist."
- NEVER end with generic "ask your doctor" — that creates ZERO engagement or traffic.

## ⚠️ STRICT CONSTRAINTS (DO NOT VIOLATE)
1. FORMAT: Output each act as a SEPARATE JSON field. Never merge into one paragraph.
2. NO SPOILERS: The answer/keyword ONLY appears in Act 3. Act 1 and Act 2 must NOT name the solution.
3. MIDROLL IS MANDATORY: Act 4 must start with "But..." / "However..." / "Here's the catch..."
4. FUNNEL CTA: Act 5 must combine "Save this" with "Comment GUIDE" or a share prompt.
5. ANALOGIES: At least one in Act 3 (e.g., "the battery pack of your eggs").
6. TONE: Warm, authoritative, empowering. NOT preachy or fear-mongering.
7. YMYL: Use hedging: "may", "research suggests", "talk to your specialist".
8. WORD COUNT: Total across all 5 acts = 55-70 words.

## ❌ FAILURE EXAMPLES (study these to avoid mistakes)

FAIL #1 — Hook spoils the answer (kills curiosity, viewer scrolls away):
  BAD hook: "Seed cycling is a myth!" → WHY FAIL: You already gave the conclusion. No reason to keep watching.
  GOOD hook: "This viral TTC trend? A fertility doctor's honest take." → Curiosity gap preserved.

FAIL #2 — Hook has no target audience (viewer doesn't feel "it's about me"):
  BAD hook: "Stop the AMH panic!" → WHY FAIL: Only people who know what AMH is will care. Too niche without context.
  GOOD hook: "Got a low AMH result and panicking? I need you to hear this." → Emotionally specific. She thinks: "That's ME."

FAIL #3 — Open loop is too generic (no tension, no stakes):
  BAD loop: "And you might be surprised by the findings." → WHY FAIL: Vague. No emotional stakes. Easy to scroll away.
  GOOD loop: "And there's one number on your blood test that's being completely misunderstood." → Specific tease with stakes.

FAIL #4 — Open loop reveals the answer (defeats its own purpose):
  BAD loop: "Most women misunderstand their AMH results." → WHY FAIL: You said "AMH" — the mystery is gone.
  GOOD loop: "One routine fertility blood test is causing more panic than it should." → Same idea, answer hidden.

FAIL #5 — CTA says "ask your doctor" (creates ZERO engagement, zero funnel traffic):
  BAD CTA: "Save this to ask your fertility specialist!" → WHY FAIL: No comment trigger, no funnel, no shareability.
  GOOD CTA: "Save this — and comment GUIDE for the full fertility checklist." → Drives comments, DM automation, LP traffic.

## ✅ SELF-CHECK (do this mentally before outputting)
Before writing your JSON, verify:
□ Does Act 1 name my TARGET AUDIENCE (age, situation, emotion)?
□ Does Act 1 create tension WITHOUT revealing the answer?
□ Does Act 2 make the viewer NEED to keep watching? Is there real stakes/mystery?
□ Does Act 2 avoid naming the main keyword/solution?
□ Does Act 3 have at least one analogy?
□ Does Act 4 start with "But..." and add genuinely NEW information?
□ Does Act 5 say "Save this video" + medical disclaimer?
If any answer is NO, rewrite that act before outputting.

## 🎯 Emphasis Words
Identify 3-5 key words to visually highlight in the video subtitles.
Pick words that are emotionally resonant or medically significant.

## 📤 Output Format
Return ONLY a valid JSON object. Each act MUST be a separate field:
{
  "hook": "Act 1 text only",
  "act2_openLoop": "Act 2 text only",
  "act3_coreInsight": "Act 3 text only",
  "act4_midrollReHook": "Act 4 text only",
  "act5_payoffCta": "Act 5 text only",
  "videoKeyword": "2-3 words for Pexels (SAFE/CALMING only, e.g. 'woman coffee')",
  "emphasisWords": ["word1", "word2", "word3"],
  "keyTakeaway": "3-6 words punchy summary of the core insight",
  "thumbnailText": "2-5 words curiosity hook for thumbnail (NO spoilers)"
}

  "keyTakeaway": "CoQ10 boosts egg quality.",
  "thumbnailText": "Your eggs need THIS."
}`;

// ── Refine a single topic ──
async function refineScript(entry: BankEntry): Promise<ScriptOutput> {
  const userPrompt = `Refine the following raw topic into a polished 20-second video script.

RAW TOPIC:
- Theme: ${entry.themeArea}
- Type: ${entry.type}
- Title: ${entry.title}
- Current Hook: ${entry.hook}
- Current Script: ${entry.text}
${entry.evidence ? `- Evidence: ${entry.evidence.summary}` : ''}
${entry.trend ? `- Trend: ${entry.trend.description} (${entry.trend.platform})` : ''}

INSTRUCTIONS:
1. Keep the hook punchy (1-5 words max). Improve it if you can.
2. Rewrite the body to be exactly 40-60 words. Make every word count.
3. Maintain medical accuracy — do NOT invent statistics or studies.
4. Suggest a Pexels video keyword (2-3 English words for vertical video search). Must be safe, calming, female-focused or abstract (e.g., "woman drinking coffee", "calm ocean", "aesthetic abstract"). Do NOT use words like "couple", "man", "intimate", "bedroom", or "romance" to avoid inappropriate or triggering B-roll.
5. Pick 3-5 emphasis words for subtitle highlighting.
6. Provide a punchy 'keyTakeaway' (3-6 words max) summarizing the main point for the final screen.
7. Provide a 'thumbnailText' (2-5 words) that creates curiosity but does NOT spoil the answer. This is for the video cover/thumbnail.
8. The script is for a DOCTOR speaking directly to camera/voiceover.

Return ONLY valid JSON.`;

  const response = await fetch('https://api.openai.com/v1/chat/completions', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${API_KEY}`,
    },
    body: JSON.stringify({
      model: 'gpt-4o',
      messages: [
        { role: 'system', content: SYSTEM_PROMPT },
        { role: 'user', content: userPrompt },
      ],
      temperature: 0.7,
      max_tokens: 500,
    }),
  });

  if (!response.ok) {
    const errorBody = await response.text();
    throw new Error(`OpenAI API Error (${response.status}): ${errorBody}`);
  }

  const result = await response.json() as any;
  const content = result.choices[0]?.message?.content?.trim();

  if (!content) {
    throw new Error('OpenAI returned empty content');
  }

  // Parse JSON (handle potential markdown fences)
  const jsonStr = content.replace(/^```json\s*\n?/, '').replace(/\n?```$/, '');
  const parsed = JSON.parse(jsonStr) as ScriptOutput;

  // Validate 5-act structure and keyTakeaway
  if (!parsed.hook || !parsed.act2_openLoop || !parsed.act3_coreInsight || !parsed.act4_midrollReHook || !parsed.act5_payoffCta || !parsed.keyTakeaway || !parsed.thumbnailText) {
    throw new Error('AI output missing required fields. Retrying may help.');
  }

  return parsed;
}

// ── Main ──
async function main() {
  const args = process.argv.slice(2);

  // Load bank
  if (!fs.existsSync(BANK_PATH)) {
    console.error('❌ topics-bank.json が見つかりません。先に npm run research:import を実行してください。');
    process.exit(1);
  }

  const bank: BankEntry[] = JSON.parse(fs.readFileSync(BANK_PATH, 'utf8'));

  // Determine which topics to refine
  let targets: BankEntry[];
  if (args.length > 0) {
    // Specific topic ID
    targets = bank.filter(e => e.id === args[0]);
    if (targets.length === 0) {
      console.error(`❌ Topic ID "${args[0]}" が見つかりません。`);
      process.exit(1);
    }
  } else {
    // All pending topics
    targets = bank.filter(e => e.status === 'pending');
    if (targets.length === 0) {
      console.log('✅ 全てのトピックが処理済みです。新しいトピックを research:import してください。');
      process.exit(0);
    }
  }

  console.log(`🎬 ${targets.length} 件のトピックを Smart Brevity で推敲します...\n`);

  for (let i = 0; i < targets.length; i++) {
    const entry = targets[i];
    console.log(`  [${i + 1}/${targets.length}] 推敲中: ${entry.title}`);

    try {
      const refined = await refineScript(entry);

      // Update the bank entry — combine acts into a single text for ElevenLabs
      const bankIndex = bank.findIndex(e => e.id === entry.id);
      if (bankIndex >= 0) {
        bank[bankIndex].hook = refined.hook;
        // Combine all acts into a flowing narration
        bank[bankIndex].text = [
          refined.act2_openLoop,
          refined.act3_coreInsight,
          refined.act4_midrollReHook,
          refined.act5_payoffCta,
        ].join(' ');
        bank[bankIndex].videoKeyword = refined.videoKeyword;
        bank[bankIndex].status = 'scripted';
        (bank[bankIndex] as any).emphasisWords = refined.emphasisWords;
        (bank[bankIndex] as any).keyTakeaway = refined.keyTakeaway;
        (bank[bankIndex] as any).thumbnailText = refined.thumbnailText;
        // Store individual acts for review
        (bank[bankIndex] as any).acts = {
          act1_hook: refined.hook,
          act2_openLoop: refined.act2_openLoop,
          act3_coreInsight: refined.act3_coreInsight,
          act4_midrollReHook: refined.act4_midrollReHook,
          act5_payoffCta: refined.act5_payoffCta,
        };
      }

      console.log(`    ✅ Act 1 (Hook):     "${refined.hook}"`);
      console.log(`    📎 Act 2 (Loop):     "${refined.act2_openLoop}"`);
      console.log(`    💡 Act 3 (Insight):  "${refined.act3_coreInsight.substring(0, 50)}..."`);
      console.log(`    🔄 Act 4 (Midroll):  "${refined.act4_midrollReHook}"`);
      console.log(`    🎯 Act 5 (CTA):      "${refined.act5_payoffCta}"`);
      console.log(`    🎨 Emphasis: ${refined.emphasisWords.join(', ')}`);
      console.log(`    📌 Key Takeaway: "${refined.keyTakeaway}"`);
      console.log(`    🖼️  Thumbnail: "${refined.thumbnailText}"`);
    } catch (error: any) {
      console.error(`    ❌ 失敗: ${error.message}`);
      continue;
    }

    // Rate limit prevention
    if (i < targets.length - 1) {
      await new Promise(r => setTimeout(r, 1500));
    }
  }

  // Save updated bank
  fs.writeFileSync(BANK_PATH, JSON.stringify(bank, null, 2));
  console.log(`\n📦 Topics Bank 更新完了`);

  // Update topics.json with scripted topics
  const scriptedTopics = bank.filter(e => e.status === 'scripted');
  const topicsForVideo = scriptedTopics.map(entry => ({
    id: entry.id,
    title: entry.title,
    hook: entry.hook,
    text: entry.text,
    videoKeyword: entry.videoKeyword,
    keyTakeaway: (entry as any).keyTakeaway || '',
    thumbnailText: (entry as any).thumbnailText || '',
  }));

  fs.writeFileSync(TOPICS_PATH, JSON.stringify(topicsForVideo, null, 2));
  console.log(`🎬 動画生成用 topics.json 更新完了（${topicsForVideo.length} 本）`);

  // ── Generate Japanese review file ──
  console.log(`\n📝 日本語確認用ファイルを作成中...`);

  const today = new Date().toISOString().split('T')[0];
  const reviewFileName = `${today}_原稿案.txt`;
  const reviewPath = path.join(process.cwd(), 'docs', reviewFileName);

  // Collect scripts that were just refined (scripted status)
  const scriptedEntries = bank.filter(e => e.status === 'scripted' && (e as any).acts);

  let reviewContent = `═══════════════════════════════════════════════════════════════
📋 原稿確認用（${today}）— ${scriptedEntries.length} 本
═══════════════════════════════════════════════════════════════

`;

  for (let i = 0; i < scriptedEntries.length; i++) {
    const entry = scriptedEntries[i];
    const acts = (entry as any).acts;
    if (!acts) continue;

    // Translate via GPT-4o
    const translationPrompt = `以下の英語ショート動画台本を日本語に翻訳してください。
各Actの構造を維持し、簡潔に翻訳してください。医学用語は正確に訳してください。
翻訳のみを返してください。説明は不要です。

Act 1 (フック): ${acts.act1_hook}
Act 2 (焦らし): ${acts.act2_openLoop}
Act 3 (核心): ${acts.act3_coreInsight}
Act 4 (再フック): ${acts.act4_midrollReHook}
Act 5 (CTA): ${acts.act5_payoffCta}`;

    let jaTranslation = '';
    try {
      const res = await fetch('https://api.openai.com/v1/chat/completions', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${API_KEY}`,
        },
        body: JSON.stringify({
          model: 'gpt-4o',
          messages: [
            { role: 'user', content: translationPrompt },
          ],
          temperature: 0.3,
          max_tokens: 500,
        }),
      });
      const data = await res.json();
      jaTranslation = data.choices?.[0]?.message?.content?.trim() || '（翻訳エラー）';
    } catch {
      jaTranslation = '（翻訳エラー）';
    }

    reviewContent += `──────────────────────────────────────
📌 ${i + 1}/${scriptedEntries.length}: ${entry.title}
──────────────────────────────────────
ID: ${entry.id}
テーマ: ${(entry as any).themeArea || '不明'}
タイプ: ${(entry as any).type || '不明'}

【英語原文（5幕構成）】
  Act 1 (フック):   ${acts.act1_hook}
  Act 2 (焦らし):   ${acts.act2_openLoop}
  Act 3 (核心):     ${acts.act3_coreInsight}
  Act 4 (再フック): ${acts.act4_midrollReHook}
  Act 5 (CTA):      ${acts.act5_payoffCta}

【日本語訳】
${jaTranslation}

【強調語】${(entry as any).emphasisWords?.join(', ') || 'なし'}
【背景動画キーワード】${entry.videoKeyword}
【エビデンス】${entry.evidence?.summary || entry.trend?.description || 'なし'}

`;

    // Rate limit
    if (i < scriptedEntries.length - 1) {
      await new Promise(r => setTimeout(r, 500));
    }
  }

  reviewContent += `═══════════════════════════════════════════════════════════════
✅ 確認完了後: npm run batch で動画生成
═══════════════════════════════════════════════════════════════
`;

  fs.writeFileSync(reviewPath, reviewContent);
  console.log(`✅ 日本語確認ファイル: docs/${reviewFileName}`);

  console.log(`\n${'═'.repeat(60)}`);
  console.log(`🎉 台本推敲完了！`);
  console.log(`${'═'.repeat(60)}`);
  console.log(`\n次のステップ:`);
  console.log(`  1. 日本語で内容確認: docs/${reviewFileName}`);
  console.log(`  2. 動画生成: npm run batch`);
}

main().catch(console.error);

// ── Main Generate Function ──
async function generateCarouselJson(entry: BankEntry, systemPrompt: string): Promise<any> {
  // Strict Evidence Rules injected from auto-generate.ts
  const evidenceRules = `
【CRITICAL MEDICAL EVIDENCE RULES (YMYL)】
1. ALL medical claims MUST be backed by high-level English literature, WHO, CDC, ACOG, ASRM, or equivalent official guidelines.
2. DO NOT use generic clinic websites, personal doctor opinions, or low-level evidence sources.
3. Slide 8 (Evidence) must explicitly cite the official source or paper.
4. Maintain a warm, empowering, but strictly objective tone. Do not use fear-mongering.
`;

  const userPrompt = `Generate a 10-slide Instagram Carousel JSON for the following raw topic.

${evidenceRules}

RAW TOPIC:
- Theme: ${entry.themeArea}
- Type: ${entry.type}
- Title: ${entry.title}
- Original Script/Hook: ${entry.text}
${entry.evidence ? `- Evidence: ${entry.evidence.summary} (${entry.evidence.source})` : ''}
${entry.trend ? `- Trend: ${entry.trend.description} (${entry.trend.platform})` : ''}

Refine this into the exact 10-slide JSON structure with the strict character limits defined in the schema.
Return ONLY valid JSON.`;

  const response = await fetch('https://api.openai.com/v1/chat/completions', {
    method: 'POST',
    headers: {

    },
    body: JSON.stringify({
      model: 'gpt-4o',
      response_format: { type: "json_object" }, // Guarantee JSON
      messages: [
        { role: 'system', content: systemPrompt },
        { role: 'user', content: userPrompt },
      ],
      temperature: 0.7,
      max_tokens: 1500,

    throw new Error(`OpenAI API Error (${response.status}): ${errorBody}`);

  return JSON.parse(content);
}

// ── Main ──
async function main() {
  const args = process.argv.slice(2);

  if (!fs.existsSync(BANK_PATH)) {
    console.error('❌ topics-bank.json が見つかりません。');
    process.exit(1);
  }

  let systemPrompt = '';
  if (fs.existsSync(PROMPT_PATH)) {
    systemPrompt = fs.readFileSync(PROMPT_PATH, 'utf8');
  } else {
    console.error(`❌ プロンプトファイルが見つかりません: ${PROMPT_PATH}`);
    process.exit(1);
  }

  if (!fs.existsSync(CAROUSEL_OUT_DIR)) {
    fs.mkdirSync(CAROUSEL_OUT_DIR, { recursive: true });
  }

  const bank: BankEntry[] = JSON.parse(fs.readFileSync(BANK_PATH, 'utf8'));

  let targets: BankEntry[];
  if (args.length > 0) {
    targets = bank.filter(e => e.id === args[0]);
    if (targets.length === 0) {
      console.error(`❌ Topic ID "${args[0]}" が見つかりません。`);
      process.exit(1);
    }
  } else {
    targets = bank.filter(e => e.carouselStatus !== 'generated');
    if (targets.length === 0) {
      console.log('✅ 全てのトピックのカルーセルが生成済みです。');
      // If we still want to generate for rendered items that don't have carouselStatus:
      const uninitialized = bank.filter(e => !e.carouselStatus);
      if (uninitialized.length > 0) {
        console.log(`🔄 carouselStatus が未設定のトピックが ${uninitialized.length} 件あります。これらを処理します。`);
        targets = uninitialized;
      } else {
        process.exit(0);
      }
    }
  }

  console.log(`📚 ${targets.length} 件のトピックからカルーセル台本を生成します...\n`);

  for (let i = 0; i < targets.length; i++) {
    const entry = targets[i];
    console.log(`  [${i + 1}/${targets.length}] 生成中: ${entry.title}`);

    try {
      const carouselData = await generateCarouselJson(entry, systemPrompt);
      
      // Enforce internal ID
      carouselData.title = entry.id;

      const outFilePath = path.join(CAROUSEL_OUT_DIR, `${entry.id}.json`);
      fs.writeFileSync(outFilePath, JSON.stringify(carouselData, null, 2));

      // Update Bank
      const bankIndex = bank.findIndex(e => e.id === entry.id);
      if (bankIndex >= 0) {
        bank[bankIndex].carouselStatus = 'generated';
      }

      console.log(`    ✅ 保存完了: ${outFilePath}`);
      console.log(`    📝 フック: "${carouselData.slides[0]?.headline}"`);

    } catch (error: any) {
      console.error(`    ❌ 失敗: ${error.message}`);
      continue;
    }

    // Rate limit
    if (i < targets.length - 1) {
      await new Promise(r => setTimeout(r, 2000));
    }
  }

  fs.writeFileSync(BANK_PATH, JSON.stringify(bank, null, 2));
  console.log(`\n📦 Topics Bank を更新しました (carouselStatus: generated)`);
}

main().catch(console.error);
